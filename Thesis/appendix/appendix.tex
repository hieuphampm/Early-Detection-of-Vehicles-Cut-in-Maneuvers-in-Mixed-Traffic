\section*{1. Báo cáo công việc - Phạm Phước Minh Hiếu}

\textbf{I. THÔNG TIN CHUNG}
\begin{itemize}
	\item \textbf{Họ và tên:} Phạm Phước Minh Hiếu \hspace{1cm} \textbf{MSSV:} 2201700085
	\item \textbf{Vai trò:} Nghiên cứu mô hình AI (YOLO, LSTM), Phát triển công cụ gán nhãn, Thiết lập logic cảnh báo \& Viết báo cáo.
	\item \textbf{Tỷ lệ đóng góp:} 50\%
\end{itemize}

\textbf{II. TỔNG QUAN NHIỆM VỤ ĐƯỢC GIAO}
\begin{itemize}
	\item Nghiên cứu tổng quan đề tài, tìm kiếm dataset, tìm hiểu kiến trúc YOLO và thuật toán theo dõi đối tượng (Deep SORT/ByteTrack).
	\item Lập trình phát triển công cụ hỗ trợ gán nhãn (Labeling tool bằng Python) và tham gia gán nhãn thủ công.
	\item Huấn luyện mô hình YOLO, thiết kế mạng phân loại ý định (LSTM) và cài đặt hệ thống luật cảnh báo tĩnh (Rule-based).
	\item Đánh giá thực nghiệm, xuất dữ liệu, soạn thảo quyển báo cáo đồ án và thiết kế slide thuyết trình.
\end{itemize}
