\textbf{I. THÔNG TIN CHUNG}
\begin{itemize}
	\item \textbf{Họ và tên:} Cao Sỹ Siêu \hspace{1cm} \textbf{MSSV:} 2201700170
	\item \textbf{Vai trò:} Sinh viên thực hiện chính (Xây dựng Dữ liệu \& Triển khai), Thực hiện gán nhãn, Thiết lập logic cảnh báo \& Viết báo cáo.
	\item \textbf{Tỷ lệ đóng góp:} 50\%
\end{itemize}

\textbf{II. TỔNG QUAN NHIỆM VỤ ĐƯỢC GIAO}
\begin{itemize}
    \item Xây dựng quy trình gán nhãn dữ liệu video dựa trên phương pháp Human-in-the-Loop (HITL) và nền tảng CVAT.
    \item Thiết lập và chuẩn hóa bộ dữ liệu thực nghiệm bao gồm 22.780 ảnh và 90.457 nhãn đối tượng mang đặc thù giao thông hỗn hợp.
    \item Huấn luyện mô hình hỗ trợ $M_{assist}$ và thực hiện tiền gán nhãn tự động để tối ưu hóa hiệu suất xây dựng dữ liệu.
    \item Thực hiện thực nghiệm, đánh giá độ trễ hệ thống trên thiết bị biên và phân tích các trường hợp sai lỗi (Error Analysis) của mô hình.
    \item Soạn thảo nội dung kỹ thuật, tổng hợp kết quả nghiên cứu vào báo cáo khóa luận và thiết kế tài liệu thuyết trình.
\end{itemize}
